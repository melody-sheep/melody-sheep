<script>
  // Add scroll effect to icons
  document.addEventListener('DOMContentLoaded', function() {
    const icons = document.querySelectorAll('.connect-icon');
    
    // Add sparkling class on hover
    icons.forEach(icon => {
      icon.addEventListener('mouseenter', function() {
        this.classList.add('sparkling');
        setTimeout(() => {
          this.classList.remove('sparkling');
        }, 1000);
      });
      
      // Add subtle float animation on scroll
      window.addEventListener('scroll', function() {
        const scrolled = window.pageYOffset;
        icons.forEach((icon, index) => {
          // Subtle floating effect based on scroll position
          const floatAmount = Math.sin((scrolled + index * 100) / 200) * 3;
          icon.style.transform = `translateY(${floatAmount}px)`;
        });
      });
    });
    
    // Random sparkling effect occasionally
    setInterval(() => {
      const randomIcon = icons[Math.floor(Math.random() * icons.length)];
      randomIcon.classList.add('sparkling');
      setTimeout(() => {
        randomIcon.classList.remove('sparkling');
      }, 800);
    }, 3000); // Sparkle every 3 seconds
  });
</script>
